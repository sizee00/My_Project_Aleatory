    <script>
      const $ = (sel) => document.querySelector(sel);

      const modalBackdrop = $("#modalBackdrop");
      const btnDetalhes = $("#btnDetalhes");
      const btnFecharModal = $("#btnFecharModal");

      const toast = $("#toast");
      let toastTimer = null;

      const btnFoco = $("#btnFoco");
      const form = $("#formCadastro");
      const nome = $("#nome");
      const email = $("#email");

      const btnTestar = $("#btnTestar");

      function showToast(message) {
        toast.textContent = message;
        toast.setAttribute("data-open", "true");
        if (toastTimer) clearTimeout(toastTimer);
        toastTimer = setTimeout(() => toast.setAttribute("data-open", "false"), 2600);
      }

      function openModal() {
        modalBackdrop.setAttribute("data-open", "true");
      }

      function closeModal() {
        modalBackdrop.setAttribute("data-open", "false");
      }

      btnDetalhes.addEventListener("click", openModal);
      btnFecharModal.addEventListener("click", closeModal);

      modalBackdrop.addEventListener("click", (e) => {
        if (e.target === modalBackdrop) closeModal();
      });

      window.addEventListener("keydown", (e) => {
        if (e.key === "Escape") closeModal();
      });

      btnFoco.addEventListener("click", () => {
        nome.focus();
      });

      btnTestar.addEventListener("click", () => {
        showToast("Notificação simulada: Lua detectada na janela de lançamento.");
      });

      form.addEventListener("submit", (e) => {
        e.preventDefault();

        const nomeVal = nome.value.trim();
        const emailVal = email.value.trim();

        if (!nomeVal) {
          showToast("Por favor, informe seu nome.");
          nome.focus();
          return;
        }

        // Validação simples
        if (!/^[^\\s@]+@[^\\s@]+\\.[^\\s@]+$/.test(emailVal)) {
          showToast("Informe um email válido.");
          email.focus();
          return;
        }

        $("#statusHint").textContent = "Cadastro recebido! Em breve enviaremos a próxima atualização.";
        showToast("Cadastro confirmado! Obrigado por participar.");
        form.reset();
        nome.focus();
      });
    </script>
